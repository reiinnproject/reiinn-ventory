    <script>
        let currentUserRole = 'staff'; 
        let inventory = JSON.parse(localStorage.getItem('rei_inv')) || [];
        let deliveries = JSON.parse(localStorage.getItem('rei_deliv')) || [];
        let gatepasses = JSON.parse(localStorage.getItem('rei_gp')) || [];
        let deployments = JSON.parse(localStorage.getItem('rei_dep')) || [];
        let schedules = JSON.parse(localStorage.getItem('rei_sched')) || [];
        let notifications = JSON.parse(localStorage.getItem('rei_notifs')) || [];
        let gpCart = [];
        let currentStatusFilter = 'All';
        let currentCalFilter = 'All';
        let editingIdx = -1;
        let selectedDepLocation = 'All';

        let currentCalDate = new Date();

        function updateClock() {
            const now = new Date();
            const timeStr = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: true });
            const dateStr = now.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: '2-digit' });
            document.getElementById('clock-time').innerText = timeStr;
            document.getElementById('clock-date').innerText = dateStr;
        }
        setInterval(updateClock, 1000);
        updateClock();

        function login() {
            const user = userInput.value;
            const pass = passInput.value;
            if(user === "admin" && pass === "admin123") { currentUserRole = 'admin'; startApp(); }
            else if(user === "staff" && pass === "staff123") { currentUserRole = 'staff'; startApp(); }
            else { err.style.display = 'block'; }
        }

        function startApp() {
            document.getElementById('login-screen').style.display = 'none';
            document.getElementById('main-app').style.display = 'flex';
            document.getElementById('live-clock').style.display = 'block'; 
            const adminElements = document.querySelectorAll('.admin-only');
            adminElements.forEach(el => {
                el.style.display = (currentUserRole === 'admin') ? 'block' : 'none';
                if(el.tagName === 'TH' || el.tagName === 'TD') el.style.display = (currentUserRole === 'admin') ? 'table-cell' : 'none';
            });
            initYearSelect();
            renderAll();
            updateNotifUI();
        }

        function toggleNotifs() {
            const dd = document.getElementById('notif-dropdown');
            dd.style.display = dd.style.display === 'block' ? 'none' : 'block';
            if(dd.style.display === 'block') clearBadge();
        }

        function clearBadge() {
            const badge = document.getElementById('notif-badge');
            badge.style.display = 'none';
            notifications.forEach(n => { if(n.for === currentUserRole) n.read = true; });
            localStorage.setItem('rei_notifs', JSON.stringify(notifications));
        }

        function clearAllNotifs() {
            notifications = notifications.filter(n => n.for !== currentUserRole);
            localStorage.setItem('rei_notifs', JSON.stringify(notifications));
            updateNotifUI();
        }

        function addNotification(msg, targetRole) {
            notifications.unshift({ msg, for: targetRole, read: false, time: new Date().toLocaleTimeString() });
            localStorage.setItem('rei_notifs', JSON.stringify(notifications));
            updateNotifUI();
        }

        function updateNotifUI() {
            const list = document.getElementById('notif-list');

        function submitGatePass() {
            const name = document.getElementById('gpReq').value;
            const loc = document.getElementById('gpLoc').value;

            renderAll();

        function initYearSelect() {
            const select = document.getElementById('calYearSelect');
            const startYear = 2024, endYear = 2030;
            for(let y=startYear; y<=endYear; y++) {
                const opt = document.createElement('option');
                opt.value = y; opt.innerText = y;
                if(y === currentCalDate.getFullYear()) opt.selected = true;
                select.appendChild(opt);
            }
        }

        function changeMonth(dir) {
            currentCalDate.setMonth(currentCalDate.getMonth() + dir);
            document.getElementById('calYearSelect').value = currentCalDate.getFullYear();
            renderCalendar();
        }

        function jumpDate() {
            currentCalDate.setFullYear(document.getElementById('calYearSelect').value);
            renderCalendar();
        }

        function renderCalendar() {
            const grid = document.getElementById('calendarGrid');
            const month = currentCalDate.getMonth(), year = currentCalDate.getFullYear();
            const firstDay = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            
            document.getElementById('calMonth').innerText = currentCalDate.toLocaleString('default', { month: 'long' });
            grid.innerHTML = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'].map(d => `<div class="calendar-day-head">${d}</div>`).join('');
            
            for(let i=0; i<firstDay; i++) grid.innerHTML += `<div></div>`;
            for(let d=1; d<=daysInMonth; d++) {
                const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
                let outEvents = [], returnEvents = [], dayScheds = [];
                if(currentCalFilter === 'All' || currentCalFilter === 'GatePass') {
                    outEvents = gatepasses.filter(g => g.status === 'Done' && g.date === dateStr);
                    returnEvents = gatepasses.filter(g => g.status === 'Done' && g.return === dateStr);
                }
                if(currentCalFilter === 'All' || currentCalFilter === 'Schedule') {
                    dayScheds = schedules.filter(s => s.date === dateStr);
                }
                let labelsHtml = '';
                dayScheds.forEach(s => { labelsHtml += `<div class="cal-label label-sched">${s.title}</div>`; });
                outEvents.forEach(e => { labelsHtml += `<div class="cal-label label-out">Out: ${e.requester}</div>`; });
                returnEvents.forEach(e => { labelsHtml += `<div class="cal-label label-return">Return: ${e.requester}</div>`; });
                grid.innerHTML += `
                    <div class="calendar-day ${(outEvents.length || returnEvents.length || dayScheds.length) ? 'bg-event' : ''}">
                        <b>${d}</b>
                        <div style="flex-grow:1; display:flex; flex-direction:column; gap:1px; margin-top:2px;">${labelsHtml}</div>
                    </div>`;
            }
        }
